clientes = []
n_clientes = 1



def menu() :
    option = int(input('''
[1] - Cadastrar cliente
[2] - Pesquisar Clientes
[3] - Mostrar todos os Clientes
[4] - Remover Clientes
[5] - Editar Cliente
[6] - Sair do programa
'''))

    return option

def cadastrar_cliente() :
    cliente_nome = input('Digite o nome do cliente: ')
    cliente_email = input('Digite o seu email: ')
    cliente_telefone = input('Digite o telefone do cliente: ')
    cliente_idade = input('Digite a sua idade: ')
    clientes_dados = (cliente_nome,cliente_email,cliente_telefone,cliente_idade)
    clientes.append(clientes_dados)
    print(clientes)
    print('Cliente adicionado')

def pesquisar_cliente() :
    for cliente in clientes:
      cliente_email = input('Digite o seu email: ')
      print(f'''
      Nome: {cliente[0]}
      email: {cliente[1]}
      Telefone: {cliente[2]}
      Idade: {cliente[3]}''')

def mostrar_cliente() :
    for cliente in clientes:
      print(f'''
      Nome: {cliente[0]}
      email: {cliente[1]}
      Telefone: {cliente[2]}
      Idade: {cliente[3]}''')

def remover_cliente() :
    for cliente in clientes:
      print(f'''
      Nome: {cliente[0]}
      email: {cliente[1]}
      Telefone: {cliente[2]}
      Idade: {cliente[3]}''')

def editar_cliente():
    for cliente in clientes:
      print(f'''
      Nome: {cliente[0]}
      email: {cliente[1]}
      Telefone: {cliente[2]}
      Idade: {cliente[3]}''')

def programa():

    while True:
        option = menu()

        if option == 1 :
            cadastrar_cliente()
        if option == 2 :
            pesquisar_cliente()
        if option == 3 :
            mostrar_cliente()
        if option == 4 :
            remover_cliente()
        if option == 5 :
            editar_cliente()
        if option == 6 :
          break

programa()
